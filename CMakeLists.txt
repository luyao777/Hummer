CMAKE_MINIMUM_REQUIRED(VERSION 3.16)
PROJECT (HUMMER LANGUAGES C CXX)
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -FMS-EXTENSIONS")
SET(CMAKE_BUILD_TYPE "DEBUG")

MESSAGE (STATUS "THIS IS BIRNARY DIR : " ${HUMMER_BINARY_DIR})
MESSAGE (STATUS "THIS IS BIRNARY/BIN DIR : " ${HUMMER_BINARY_DIR}/BIN)
MESSAGE (STATUS "THIS IS SOURCE DIR : " ${HUMMER_SOURCE_DIR})

ADD_SUBDIRECTORY(HUMMER)

SET (SRC_LIST SANDBOX/APPLICATION.CPP)
ADD_EXECUTABLE(APPLICATION ${SRC_LIST})
TARGET_LINK_LIBRARIES(APPLICATION HUMMERLIB)
